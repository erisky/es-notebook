From skysoft/product/hardware/ap5400w/meta-scripts/askey-ap5400w/nand-flash.conf
The single image contains:
[mibib]   ->  partition table 
          ->  generated by:
            python genimg.py --partition_tool=partition_tool --mbn_gen=nand_mbn_generator.py --skip_export -- image_name=NAND_IMAGES
            (our way: IPQ4019.ILQ.5.0/common/build/ipq/tools/askey-ap5400w-gen.sh)

[sbl1]    -> SBL - Seconday boot Loader code
$$ partition = 0:SBL1
          -> filename = sbl1_nand.mbn   from chipcode:
                BOOT.BF.3.1.1/boot_images/build/ms/bin/40xx/nand/sbl1_nand.mbn
[bootconfig]
          -> as above genimg.py
$$ partition = 0:BOOTCONFIG
            filename = nand_bootconfig.bin   -> askey-ap5400w-gen.sh -> genimg.py

[bootconfig1]
          -> as above genimg.py
$$ partition = 0:BOOTCONFIG1
            filename = nand_bootconfig1.bin   -> askey-ap5400w-gen.sh -> genimg.py


[ddr-AP5400W-REV1]
partition = 0:CDT
filename = cdt-pcddr_AP5400W-REV1.bin    ---> renamed in askey-${hardware}-gen.sh   ::> why not just uses shared config
if_machid = 0x8540001                    

[tz]
partition = 0:QSEE
filename = tz.mbn  --> from chipcode: ./TZ.BF.2.7/trustzone_images/build/ms/bin/MAZAANAA/tz.mbn

[u-boot]
partition = 0:APPSBL
filename = openwrt-ipq40xx-u-boot-stripped.elf

[ubi]
partition = rootfs
filename = openwrt-ipq806x-ap5400w-ubi-root.img

[userdata]
partition = ubifs
filename = ubi-fs.img

NAND_IMAGES NOR_IMAGES EMMC_IMAGES NOR_PLUS_NAND_IMAGES SMEM_IMAGES CDT_IMAGES BOOTCONFIG_IMAGES
--> askey regenerate all by askey-${hardware}-gen.sh
